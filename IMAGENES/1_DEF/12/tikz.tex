\documentclass{standalone}
\usepackage{tikz}

\begin{document}

\newcommand{\hex}[1]{
	\draw (#1) \foreach \n in {1,...,6}{--++ (\n*60:1cm)};
	% \fill (#1) ++ (120:1cm) circle (1.5pt);
	}

\begin{tikzpicture}[>=latex]
	\def\r{1.73205}
	\foreach \n in {1,...,3} {\hex{90:\n*\r}}
	\foreach \n in {1,...,3} {\hex{1.5,\n*\r+\r/2}}
	\foreach \n in {1,...,3} {\hex{-1.5,\n*\r+\r/2}}
	\foreach \n in {1,...,3} {\hex{3,\n*\r}}
	\foreach \n in {1,...,3} {\hex{-3,\n*\r}}
\end{tikzpicture}

\end{document}
